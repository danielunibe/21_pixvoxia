import React, { useEffect, useRef, useState } from 'react';




// --- utils ---

const lerp = (a, b, t) => a + (b - a) * t;

const clamp = (v, min, max) => Math.max(min, Math.min(max, v));




// --- easings ---

const easeInOutCubic = (t) =>

  t < 0.5 ? 4 * t * t * t : 1 - Math.pow(-2 * t + 2, 3) / 2;




// overshoot suave (no caricatura)

const easeOutBackSoft = (t, s = 1.15) =>

  1 + (s + 1) * Math.pow(t - 1, 3) + s * Math.pow(t - 1, 2);




// asentado amortiguado muy sutil

const dampedSettle = (t) => {

  const freq = 2.0;

  const decay = 5.0;

  return 1 - Math.exp(-decay * t) * Math.cos(2 * Math.PI * freq * t);

};




// --- geometry helpers ---

const interpolatePoints = (from, to, t) =>

  from.map(([x1, y1], i) => {

    const [x2, y2] = to[i];

    return [lerp(x1, x2, t), lerp(y1, y2, t)];

  });




const pointsToPath = (pts) =>

  pts.map(([x, y], i) => `${i === 0 ? 'M' : 'L'} ${x} ${y}`).join(' ') + ' Z';




const pointsToLine = (pts) =>

  pts.map(([x, y], i) => `${i === 0 ? 'M' : 'L'} ${x} ${y}`).join(' ');




// --- geometry ---

const TOP_3D = [[200,210],[286.6,160],[200,110],[113.4,160]];

const LEFT_3D = [[200,210],[113.4,160],[113.4,260],[200,310]];

const RIGHT_3D = [[200,210],[286.6,160],[286.6,260],[200,310]];




const TOP_2D = [[200,286.6],[286.6,200],[200,113.4],[113.4,200]];

const LEFT_2D = [[200,286.6],[113.4,200],[113.4,200],[200,286.6]];

const RIGHT_2D = [[200,286.6],[286.6,200],[286.6,200],[200,286.6]];




const HL1_3D = [[200,210],[113.4,160]];

const HL2_3D = [[200,210],[286.6,160]];

const HL3_3D = [[200,210],[200,310]];




const HL1_2D = [[200,286.6],[113.4,200]];

const HL2_2D = [[200,286.6],[286.6,200]];

const HL3_2D = [[200,286.6],[200,286.6]];




const App = () => {

  const [isTopView, setIsTopView] = useState(false);

  const [progress, setProgress] = useState({ geo: 0, phys: 0 });




  const progressRef = useRef(0);

  const rafRef = useRef(null);

  const hasInteractedRef = useRef(false);




  // --- animation driver ---

  const animateTo = (target) => {

    if (rafRef.current) cancelAnimationFrame(rafRef.current);




    const from = progressRef.current;

    const duration = 780;

    const start = performance.now();




    const tick = (now) => {

      const raw = clamp((now - start) / duration, 0, 1);




      // geometría estable

      const geoT = easeInOutCubic(raw);




      // física perceptual

      let physT;




      // anticipación

      if (raw < 0.12) {

        const t = raw / 0.12;

        physT = -0.04 * (1 - t);

      }

      // avance + overshoot sutil

      else if (raw < 0.78) {

        const t = (raw - 0.12) / 0.66;

        physT = easeOutBackSoft(t) * 1.01;

      }

      // asentado amortiguado

      else {

        const t = (raw - 0.78) / 0.22;

        physT = 1 + (dampedSettle(t) - 1) * 0.06;

      }




      const geoValue = lerp(from, target, geoT);




      progressRef.current = geoValue;

      setProgress({ geo: geoValue, phys: physT });




      if (raw < 1) rafRef.current = requestAnimationFrame(tick);

    };




    rafRef.current = requestAnimationFrame(tick);

  };




  // --- interaction ---

  const toggleView = () => {

    hasInteractedRef.current = true;

    setIsTopView((prev) => !prev);

  };




  useEffect(() => {

    if (!hasInteractedRef.current) return;

    animateTo(isTopView ? 1 : 0);




    return () => {

      if (rafRef.current) cancelAnimationFrame(rafRef.current);

    };

  }, [isTopView]);




  const { geo, phys } = progress;




  // --- paths ---

  const topPath = pointsToPath(interpolatePoints(TOP_3D, TOP_2D, geo));

  const leftPath = pointsToPath(interpolatePoints(LEFT_3D, LEFT_2D, geo));

  const rightPath = pointsToPath(interpolatePoints(RIGHT_3D, RIGHT_2D, geo));

  const translateY = lerp(0, -9, phys) + Math.sin(geo * Math.PI * 2) * 0.8;




  const highlightOpacity = lerp(1, 0, geo);




  return (

          viewBox="0 0 400 400"

          className="w-80 h-80"

          style={{

            transform: `translateY(${translateY}px) scale(${scaleX}, ${scaleY})`,

            transformOrigin: 'center',

            willChange: 'transform',

          }}

        >

          <defs>

            <linearGradient id="topFace" x1="0%" y1="0%" x2="100%" y2="100%">

              <stop offset="0%" stopColor="#4fd1c5" />

              <stop offset="100%" stopColor="#319795" />

            </linearGradient>




            <linearGradient id="leftFace" x1="0%" y1="0%" x2="100%" y2="100%">

              <stop offset="0%" stopColor="#285e61" />

              <stop offset="100%" stopColor="#234e52" />

            </linearGradient>




            <linearGradient id="rightFace" x1="0%" y1="0%" x2="0%" y2="100%">

              <stop offset="0%" stopColor="#319795" />

              <stop offset="100%" stopColor="#285e61" />

            </linearGradient>

          </defs>




          {/* caras */}

          <path d={leftPath} fill="url(#leftFace)" />

          <path d={rightPath} fill="url(#rightFace)" />

          <path d={topPath} fill="url(#topFace)" />




          {/* detalles */}

          <path d={hl1} stroke="rgba(255,255,255,0.35)" strokeWidth={1.2} fill="none" style={{ opacity: highlightOpacity }} />

          <path d={hl2} stroke="rgba(255,255,255,0.22)" strokeWidth={1.2} fill="none" style={{ opacity: highlightOpacity * 0.9 }} />

          <path d={hl3} stroke="rgba(0,0,0,0.18)" strokeWidth={1.2} fill="none" style={{ opacity: highlightOpacity * 0.85 }} />

        </svg>

      </div>

    </div>

  );

};
